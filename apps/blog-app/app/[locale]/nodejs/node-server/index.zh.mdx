# Node Server

``` JavaScript
const http = require('node:http')

// 每当请求到达服务器时都会执行此回调函数
const server = http.createServer((req, res) => {
    res.writeHead(200, {'Content-type': "text/plain"});
    res.end('hello world');
});

server.listen(3000, () => {
    console.log('server running on port 3000')
});
```

## JSON Response

``` JavaScript
const http = require('node:http')

// 每当请求到达服务器时都会执行此回调函数
const server = http.createServer((req, res) => {

    const superHero = {
        "firstName": "aaa"
    }

    res.writeHead(200, {'Content-type': "application/json"});
    res.end(JSON.stringify(superHero));
});

server.listen(3000, () => {
    console.log('server running on port 3000')
});
```

## HTML Response

``` JavaScript
const http = require('node:http')

// 每当请求到达服务器时都会执行此回调函数
const server = http.createServer((req, res) => {
    res.writeHead(200, {'Content-type': "text/html"});
    res.end("<h1>hello world</h1>");
});

server.listen(3000, () => {
    console.log('server running on port 3000')
});
```

html作为单独的一份文件

`xxx.html`:

``` html
<!DOCTYPE HTML>
<HTML>
    <head>
        <title>xxx.html</title>
    </head>
    <body>
        <h1>hello world</h1>
    </body>
</HTML>
```

``` JavaScript
const http = require('node:http');
const fs = require('node:fs');

// 每当请求到达服务器时都会执行此回调函数
const server = http.createServer((req, res) => {
    res.writeHead(200, {'Content-type': "text/html"});
    const html = fs.readFileSync('./xxx.html');
    res.end(html);
});

server.listen(3000, () => {
    console.log('server running on port 3000')
});
```
